create_overdue_dose_logs

"
DECLARE
  med_record RECORD;
  schedule_record RECORD;
  today_date DATE;
  user_tz TEXT;
  user_current_time TIME;
BEGIN
  today_date := CURRENT_DATE::DATE;

  -- Loop through all active medications
  FOR med_record IN
    SELECT m.id, m.user_id, m.name, m.form, m.dosage_amount, m.dosage_unit, p.timezone
    FROM medications m
    LEFT JOIN profiles p ON m.user_id = p.id
    WHERE m.is_active = true
      AND m.start_date <= today_date
      AND (m.end_date IS NULL OR m.end_date >= today_date)
  LOOP
    -- Get user's timezone (default to Asia/Karachi if not set)
    user_tz := COALESCE(med_record.timezone, 'Asia/Karachi');
    
    -- Convert current UTC time to user's timezone
    user_current_time := (NOW() AT TIME ZONE user_tz)::TIME;
    
    -- Loop through each schedule for this medication
    FOR schedule_record IN
      SELECT ms.time_to_take
      FROM medication_schedules ms
      WHERE ms.medication_id = med_record.id
    LOOP
      -- Check if time has passed (using user's local time)
      IF schedule_record.time_to_take <= user_current_time THEN
        -- Insert dose_log if it doesn't already exist
        INSERT INTO dose_logs (
          user_id,
          scheduled_date,
          scheduled_time,
          status,
          medication_name,
          medication_form,
          dosage_amount,
          dosage_unit
        )
        SELECT
          med_record.user_id,
          today_date,
          schedule_record.time_to_take,
          'Pending',
          med_record.name,
          med_record.form,
          med_record.dosage_amount,
          med_record.dosage_unit
        WHERE NOT EXISTS (
          SELECT 1 FROM dose_logs
          WHERE user_id = med_record.user_id
            AND scheduled_date = today_date
            AND scheduled_time = schedule_record.time_to_take
            AND medication_name = med_record.name
        );
      END IF;
    END LOOP;
  END LOOP;
END;
"

get_followup_reminders

"
DECLARE
  v_email TEXT;
  v_email_alerts BOOLEAN;
BEGIN
  FOR v_email, v_email_alerts IN
    SELECT p.email, p.email_alerts FROM profiles p
  LOOP
  END LOOP;

  RETURN QUERY
  SELECT
    dl.id,
    p.email,
    p.full_name,
    p.patient_id,
    dl.medication_name,
    dl.medication_form,
    dl.dosage_amount,
    dl.dosage_unit,
    dl.scheduled_date,
    dl.scheduled_time,
    dl.reminder_sent_at,
    dl.action_token,
    p.timezone
  FROM dose_logs dl
  JOIN profiles p ON p.id = dl.user_id
  WHERE
    dl.status = 'Pending'
    AND dl.followup_sent_at IS NULL
    AND (dl.scheduled_date + dl.scheduled_time) < ((NOW() AT TIME ZONE p.timezone) - INTERVAL '1 hour')
    AND p.email_alerts = TRUE;
END;
"

get_medications_due_soon

"
DECLARE
  rec RECORD;
  v_now            TIMESTAMPTZ;
  v_today          DATE;
  v_current_time   TIME;
  v_window_end     TIMESTAMPTZ;
  v_spans_midnight BOOLEAN;
  v_dow            INTEGER;
  v_dom            INTEGER;
BEGIN
  -- Loop per user so each gets their own timezone applied
  FOR rec IN
    SELECT DISTINCT m.user_id, p.timezone
    FROM medications m
    JOIN profiles p ON p.id = m.user_id
    WHERE m.is_active = TRUE
  LOOP
    -- Calculate time variables in THIS user's timezone
    v_now            := NOW() AT TIME ZONE rec.timezone;
    v_today          := v_now::DATE;
    v_current_time   := v_now::TIME;
    v_window_end     := (NOW() + (minutes_ahead || ' minutes')::INTERVAL) AT TIME ZONE rec.timezone;
    v_spans_midnight := v_window_end::DATE > v_today;
    v_dow            := EXTRACT(DOW FROM v_today)::INTEGER;
    v_dom            := EXTRACT(DAY FROM v_today)::INTEGER;

    RETURN QUERY
    SELECT
      m.id                                                      AS medication_id,
      p.email                                                   AS user_email,
      p.full_name                                               AS user_name,
      p.patient_id                                             AS patient_id,
      m.name                                                    AS medication_name,
      m.form,
      m.dosage_amount,
      m.dosage_unit,
      m.amount_per_time,
      m.consumption_unit,
      m.dose_amount_left,
      m.stock_unit,
      ms.time_label,
      ms.time_to_take,
      m.notes,
      m.prescribed_by,
      m.frequency_details->>'type'                             AS frequency_type,
      COALESCE(ms.time_label, '') || ' ├втВмтАЬ ' ||
        TO_CHAR(ms.time_to_take, 'HH12:MI AM')                AS scheduled_at,
      rec.timezone                                             AS user_timezone

    FROM medications m
    JOIN medication_schedules ms ON ms.medication_id = m.id
    JOIN profiles p              ON p.id = m.user_id

    WHERE
      m.user_id = rec.user_id

      AND m.is_active = TRUE
      AND m.start_date <= v_today
      AND (m.end_date IS NULL OR m.end_date >= v_today)

      AND p.email_alerts = TRUE

      /*├втАЭтВм├втАЭтВм TIME WINDOW ├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм*/
      AND (
        minutes_ahead >= 1440

        OR (
          NOT v_spans_midnight
          AND ms.time_to_take >= v_current_time
          AND ms.time_to_take <  v_window_end::TIME
        )

        OR (
          v_spans_midnight
          AND (
            ms.time_to_take >= v_current_time
            OR ms.time_to_take < v_window_end::TIME
          )
        )
      )

      /*├втАЭтВм├втАЭтВм FREQUENCY LOGIC ├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм├втАЭтВм*/
      AND (

        -- CASE 1: DAILY
        (m.frequency_details->>'type' = 'daily')

        -- CASE 2: WEEKLY
        OR (
          m.frequency_details->>'type' = 'weekly'
          AND (m.frequency_details->'selectedDays') IS NOT NULL
          AND (m.frequency_details->'selectedDays') @> to_jsonb(v_dow)
        )

        -- CASE 3: INTERVAL
        OR (
          m.frequency_details->>'type' = 'interval'
          AND (m.frequency_details->>'intervalDays') IS NOT NULL
          AND (m.frequency_details->>'intervalDays')::INTEGER > 0
          AND MOD(
                (v_today - m.start_date),
                (m.frequency_details->>'intervalDays')::INTEGER
              ) = 0
        )

        -- CASE 4: MONTHLY by specific dates
        OR (
          m.frequency_details->>'type'            = 'monthly'
          AND m.frequency_details->>'monthlyMode' = 'dates'
          AND (m.frequency_details->'selectedMonthDates') IS NOT NULL
          AND EXISTS (
            SELECT 1
            FROM jsonb_array_elements_text(
                   m.frequency_details->'selectedMonthDates'
                 ) AS d(val)
            WHERE regexp_replace(d.val, '[^0-9]', '', 'g')::INTEGER = v_dom
          )
        )

        -- CASE 5: MONTHLY by count
        OR (
          m.frequency_details->>'type'            = 'monthly'
          AND m.frequency_details->>'monthlyMode' = 'count'
          AND (m.frequency_details->>'monthCount') IS NOT NULL
          AND (m.frequency_details->>'monthCount')::INTEGER > 0
          AND EXISTS (
            SELECT 1
            FROM generate_series(
                   0,
                   (m.frequency_details->>'monthCount')::INTEGER - 1
                 ) AS gs(i)
            CROSS JOIN LATERAL (
              SELECT DATE_PART('days',
                DATE_TRUNC('month', v_today) + INTERVAL '1 month - 1 day'
              )::INTEGER AS days_in_month
            ) dim
            WHERE ROUND(
                    (gs.i + 0.5)
                    * dim.days_in_month
                    / (m.frequency_details->>'monthCount')::INTEGER
                  )::INTEGER = v_dom
          )
        )

        -- CASE 6: LEGACY FALLBACK
        OR (
          m.frequency_details IS NULL
          AND m.frequency IN (
            'daily', 'Once Daily', 'Twice Daily', 'Three Times Daily'
          )
        )

      );
  END LOOP;
END;

rls_auto_enable

"
DECLARE
  cmd record;
BEGIN
  FOR cmd IN
    SELECT *
    FROM pg_event_trigger_ddl_commands()
    WHERE command_tag IN ('CREATE TABLE', 'CREATE TABLE AS', 'SELECT INTO')
      AND object_type IN ('table','partitioned table')
  LOOP
     IF cmd.schema_name IS NOT NULL AND cmd.schema_name IN ('public') AND cmd.schema_name NOT IN ('pg_catalog','information_schema') AND cmd.schema_name NOT LIKE 'pg_toast%' AND cmd.schema_name NOT LIKE 'pg_temp%' THEN
      BEGIN
        EXECUTE format('alter table if exists %s enable row level security', cmd.object_identity);
        RAISE LOG 'rls_auto_enable: enabled RLS on %', cmd.object_identity;
      EXCEPTION
        WHEN OTHERS THEN
          RAISE LOG 'rls_auto_enable: failed to enable RLS on %', cmd.object_identity;
      END;
     ELSE
        RAISE LOG 'rls_auto_enable: skip % (either system schema or not in enforced list: %.)', cmd.object_identity, cmd.schema_name;
     END IF;
  END LOOP;
END;
"

update_missed_doses

"
BEGIN
  UPDATE dose_logs dl
  SET status = 'Missed'
  FROM profiles p
  WHERE dl.user_id = p.id
    AND dl.scheduled_date = CURRENT_DATE::DATE
    AND dl.status = 'Pending'
    AND (NOW() AT TIME ZONE COALESCE(p.timezone, 'Asia/Karachi'))::TIME >= (dl.scheduled_time + INTERVAL '2 hours')::TIME;
END;
"
