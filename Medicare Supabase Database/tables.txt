profiles table schema

create table public.profiles (
  id uuid not null,
  patient_id text not null default (
    'P-'::text || upper(
      "substring" ((extensions.uuid_generate_v4 ())::text, 1, 8)
    )
  ),
  full_name text not null,
  email text not null,
  date_of_birth date null,
  gender text null,
  known_conditions text null,
  allergies text null,
  blood_type text null,
  organ_donor boolean null default false,
  weight_kg numeric(5, 2) null,
  height_cm numeric(5, 2) null,
  emergency_contact_name text null,
  emergency_contact_number text null,
  emergency_contact_relation text null,
  profile_completed boolean null default false,
  created_at timestamp with time zone null default now(),
  updated_at timestamp with time zone null default now(),
  email_alerts boolean null default true,
  ai_assistant boolean not null default false,
  timezone text not null default 'Asia/Karachi'::text,
  constraint profiles_pkey primary key (id),
  constraint profiles_patient_id_key unique (patient_id),
  constraint profiles_id_fkey foreign KEY (id) references auth.users (id) on delete CASCADE,
  constraint profiles_blood_type_check check (
    (
      blood_type = any (
        array[
          'A+'::text,
          'A-'::text,
          'B+'::text,
          'B-'::text,
          'O+'::text,
          'O-'::text,
          'AB+'::text,
          'AB-'::text,
          'Unknown'::text
        ]
      )
    )
  ),
  constraint profiles_gender_check check (
    (
      gender = any (
        array[
          'Male'::text,
          'Female'::text,
          'Other'::text,
          'Prefer not to say'::text
        ]
      )
    )
  )
) TABLESPACE pg_default;

create trigger trg_profiles_updated_at BEFORE
update on profiles for EACH row
execute FUNCTION update_updated_at ();
